185 void next(State& state) const {
186 if (!state.bad.empty() && state.bad.front() == state.position) {
187 state.bad.pop_front();
188 if (state.bad.empty()) {
189 // This should effectively clear the ambiguous bitset, allowing
190 // us to skip its shifting if there are no more ambiguous
191 // bases. We do it here because we won't get an opportunity to
192 // do it later; as 'bad' is empty, the shift below is skipped.
193 shift(state.ambiguous);
194 }
195 }
196

198 char base = state.seq[right];
199 if (is_good(base)) {
200 add_base(state.state, base); // no need to trim off the end, the mask will handle that.

203 }
204 } else {

207 state.bad.push_back(right);

235 static int strand_match(const State& match, const std::bitset<N>& ref, const std::bitset<N>& mask) {
236 // pop count here is equal to the number of non-ambiguous mismatches *
237 // 2 + number of ambiguous mismatches * 3. This is because
238 // non-ambiguous bases are encoded by 1 set bit per 4 bases (so 2 are
239 // left after a XOR'd mismatch), while ambiguous mismatches are encoded
240 // by all set bits per 4 bases (which means that 3 are left after XOR).
241 int pcount = ((match.state & mask) ^ ref).count();
242
243 // Counting the number of ambiguous bases after masking. Each ambiguous
244 // base is represented by 4 set bits, so we divide by 4 to get the number
245 // of bases; then we multiply by three to remove their contribution. The
246 // difference is then divided by two to get the number of non-ambig mm's.
247 if (!match.bad.empty()) {
248 int acount = (match.ambiguous & mask).count();
249 acount /= 4;
250 return acount + (pcount - acount * 3) / 2;
251 } else {
252 return pcount / 2;
253 }
254 }
