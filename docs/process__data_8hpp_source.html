123void process_single_end_data(Pointer_ input, Handler_& handler, const ProcessSingleEndDataOptions& options) {
124 FastqReader<Pointer_> fastq(input);
125 bool finished = false;
126
127 std::vector<ChunkOfReads> reads(options.num_threads);
128 std::vector<std::thread> jobs(options.num_threads);
129 std::vector<decltype(handler.initialize())> states(options.num_threads);
130 std::vector<std::string> errs(options.num_threads);
131
132 auto join = [&](int i) -> void {
133 if (jobs[i].joinable()) {
134 jobs[i].join();
135 if (errs[i] != "") {
136 throw std::runtime_error(errs[i]);
137 }
138 handler.reduce(states[i]);
139 reads[i].clear(Handler_::use_names);
140 }
141 };
142
143 // Safety measure to enforce const-ness within each thread.
144 const Handler_& conhandler = handler;
145
146 try {
147 while (!finished) {
148 for (int t = 0; t < options.num_threads; ++t) {
149 join(t);
150
151 auto& curreads = reads[t];
152 for (int b = 0; b < options.block_size; ++b) {
153 if (!fastq()) {
154 finished = true;
155 break;
156 }
157
158 curreads.add_read_sequence(fastq.get_sequence());
159 if constexpr(Handler_::use_names) {
160 curreads.add_read_name(fastq.get_name());
161 }
162 }
163
164 states[t] = conhandler.initialize();
165 jobs[t] = std::thread([&](int i) -> void {
166 try {
167 auto& state = states[i];
168 const auto& curreads = reads[i];
169 size_t nreads = curreads.size();
170
171 if constexpr(!Handler_::use_names) {
172 for (size_t b = 0; b < nreads; ++b) {
173 conhandler.process(state, curreads.get_sequence(b));
174 }
175 } else {
176 for (size_t b = 0; b < nreads; ++b) {
177 conhandler.process(state, curreads.get_name(b), curreads.get_sequence(b));
178 }
179 }
180 } catch (std::exception& e) {
181 errs[i] = std::string(e.what());
182 }
183 }, t);
184
185 if (finished) {
186 // We won't get a future iteration to join the previous threads
187 // that we kicked off, so we do it now.
188 for (int u = 0; u < options.num_threads; ++u) {
189 auto pos = (u + t + 1) % options.num_threads;
190 join(pos);
191 }
192 break;
193 }
194 }
195 }
196 } catch (std::exception& e) {
197 // Mopping up any loose threads, so to speak.
198 for (int t = 0; t < options.num_threads; ++t) {
199 if (jobs[t].joinable()) {
200 jobs[t].join();
201 }
202 }
203 throw;
204 }
205
206 return;
207}
