80class ThreadPool {
81public:
82 template<typename RunJob_>
83 ThreadPool(RunJob_ run_job, int num_threads) : my_helpers(num_threads) {
84 my_threads.reserve(num_threads);
85 for (int t = 0; t < num_threads; ++t) {
86 // Copy lambda as it will be gone once this constructor finishes.
87 my_threads.emplace_back([run_job,this](int thread) -> void {
88 auto& env = my_helpers[thread];
89 while (1) {
90 std::unique_lock lck(env.mut);
91 env.cv.wait(lck, [&]() -> bool { return env.input_ready; });
92 if (env.terminated) {
93 return;
94 }
95 env.input_ready = false;
96
97 try {
98 run_job(env.work);
99 } catch (...) {
100 std::lock_guard elck(my_error_mut);
101 if (!my_error) {
102 my_error = std::current_exception();
103 }
104 }
105
106 env.has_output = true;
107 env.available = true;
108 env.cv.notify_one();
109 }
110 }, t);
111 }
112 }
113
114 ~ThreadPool() {
115 for (auto& env : my_helpers) {
116 {
117 std::lock_guard lck(env.mut);
118 env.terminated = true;
119 env.input_ready = true;
120 }
121 env.cv.notify_one();
122 }
123 for (auto& thread : my_threads) {
124 thread.join();
125 }
126 }
127
128private:
129 std::vector<std::thread> my_threads;
130
131 struct Helper {
132 std::mutex mut;
133 std::condition_variable cv;
134 bool input_ready = false;
135 bool available = true;
136 bool has_output = false;
137 bool terminated = false;
138 Workspace_ work;
139 };
140 std::vector<Helper> my_helpers;
141
142 std::mutex my_error_mut;
143 std::exception_ptr my_error;
144
145public:
146 template<typename CreateJob_, typename MergeJob_>
147 void run(CreateJob_ create_job, MergeJob_ merge_job) {
148 auto num_threads = my_threads.size();
149 bool finished = false;
150 decltype(num_threads) thread = 0, finished_count = 0;
151
152 // We submit jobs by cycling through all threads, then we merge their results in order of submission.
153 // This is a less efficient worksharing scheme but it guarantees the same order of merges.
154 while (1) {
155 auto& env = my_helpers[thread];
156 std::unique_lock lck(env.mut);
157 env.cv.wait(lck, [&]() -> bool { return env.available; });
158
159 {
160 std::lock_guard elck(my_error_mut);
161 if (my_error) {
162 std::rethrow_exception(my_error);
163 }
164 }
165 env.available = false;
166
167 if (env.has_output) {
168 merge_job(env.work);
169 env.has_output = false;
170 }
171
172 if (finished) {
173 // Go through all threads one last time, making sure all results are merged.
174 ++finished_count;
175 if (finished_count == num_threads) {
176 break;
177 }
178 } else {
179 // 'create_job' is responsible for parsing the FASTQ file and
180 // creating a ChunkOfReads. Unfortunately I can't figure out
181 // how to parallelize the parsing as it is very hard to jump
182 // into a middle of a FASTQ file and figure out where the next
183 // entry is; this is because (i) multiline sequences are legal
184 // and (ii) +/@ are not sufficient delimiters when they can
185 // show up in the quality scores.
186 finished = create_job(env.work);
187 env.input_ready = true;
188 lck.unlock();
189 env.cv.notify_one();
190 }
191
192 ++thread;
193 if (thread == num_threads) {
194 thread = 0;
195 }
196 }
197 }
