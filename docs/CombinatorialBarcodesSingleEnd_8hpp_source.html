114 std::pair<bool, int> find_match(
115 const char* seq,
116 size_t position,
117 int obs_mismatches,
118 const std::array<SimpleBarcodeSearch, num_variable>& libs,
119 std::array<typename SimpleBarcodeSearch::State, num_variable>& states,
120 std::array<int, num_variable>& temp)
121 const {
122 const auto& regions = constant_matcher.template variable_regions<reverse>();
123
124 for (size_t r = 0; r < num_variable; ++r) {
125 auto range = regions[r];
126 auto start = seq + position;
127 std::string current(start + range.first, start + range.second);
128
129 auto& curstate = states[r];
130 libs[r].search(current, curstate, max_mm - obs_mismatches);
131 if (curstate.index < 0) {
132 return std::make_pair(false, 0);
133 }
134
135 obs_mismatches += curstate.mismatches;
136 if (obs_mismatches > max_mm) {
137 return std::make_pair(false, 0);
138 }
139
140 if constexpr(reverse) {
141 temp[num_variable - r - 1] = curstate.index;
142 } else {
143 temp[r] = curstate.index;
144 }
145 }
146
147 return std::make_pair(true, obs_mismatches);
148 }

159 void process_first(State& state, const std::pair<const char*, const char*>& x) const {

181 void process_best(State& state, const std::pair<const char*, const char*>& x) const {
182 auto deets = constant_matcher.initialize(x.first, x.second - x.first);
183 bool found = false;
184 int best_mismatches = max_mm + 1;
185 std::array<int, num_variable> best_id;
186
187 auto update = [&](std::pair<int, int> match) -> void {
188 if (match.first && match.second <= best_mismatches) {
189 if (match.second == best_mismatches) {
190 if (best_id != state.temp) { // ambiguous.
191 found = false;
192 }
193 } else {
194 // A further optimization at this point would be to narrow
195 // max_mm to the current 'best_mismatches'. But
196 // this probably isn't worth it.
197
198 found = true;
199 best_mismatches = match.second;
200 best_id = state.temp;
201 }
202 }
203 };
204
205 while (!deets.finished) {
206 constant_matcher.next(deets);
207
208 if (forward && deets.forward_mismatches <= max_mm) {
209 update(forward_match(x.first, deets, state));
210 }
211
212 if (reverse && deets.reverse_mismatches <= max_mm) {
213 update(reverse_match(x.first, deets, state));
214 }
215 }
216
217 if (found) {
218 state.collected.push_back(best_id);
219 }
220 }
