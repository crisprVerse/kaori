26inline void fill_library(const std::vector<const char*>& options, std::unordered_map<std::string, BarcodeIndex>& exact, Trie_& trie, bool reverse) {
27 std::size_t len = trie.length();
28 auto nopt = options.size();
29
30 for (decltype(nopt) i = 0; i < nopt; ++i) {
31 auto ptr = options[i];
32
33 std::string current;
34 if (!reverse) {
35 current = std::string(ptr, ptr + len);
36 } else {
37 current.reserve(len);
38 for (size_t j = 0; j < len; ++j) {
39 current += complement_base<true, true>(ptr[len - j - 1]);
40 }
41 }
42
43 // Note that this must be called, even if the sequence is duplicated;
44 // otherwise the trie's internal counter will not be properly incremented.
45 auto status = trie.add(current.c_str());
46
47 if (!status.has_ambiguous) {
48 if (!status.is_duplicate || status.duplicate_replaced) {
49 exact[current] = i;
50 } else if (status.duplicate_cleared) {
51 exact[current] = STATUS_UNMATCHED;
52 }
53 }
54 }
55
56 trie.optimize();
57 return;
58}

201 void search(const std::string& search_seq, State& state, int allowed_mismatches) const {
202 auto it = my_exact.find(search_seq);
203 if (it != my_exact.end()) {
204 state.index = it->second;
205 state.mismatches = 0;
206 return;
207 }
208
209 auto set_from_cache = [&](const CacheEntry& cached) -> void {
210 if (cached.mismatches > allowed_mismatches) {
211 state.index = STATUS_UNMATCHED;
212 state.mismatches = 0;
213 } else {
214 state.index = cached.index;
215 state.mismatches = cached.mismatches;
216 }
217 };
218
219 auto cIt = my_cache.find(search_seq);
220 if (cIt != my_cache.end()) {
221 set_from_cache(cIt->second);
222 return;
223 }
224
225 auto lIt = state.cache.find(search_seq);
226 if (lIt != state.cache.end()) {
227 set_from_cache(lIt->second);
228 return;
229 }
230
231 auto missed = my_trie.search(search_seq.c_str(), allowed_mismatches);
232 if (is_barcode_index_ok(missed.index)) {
233 // No need to check against allowed_mismatches, as we explicitly searched for that in the trie.
234 state.index = missed.index;
235 state.mismatches = missed.mismatches;
236 state.cache[search_seq] = CacheEntry(missed.index, missed.mismatches);
237 return;
238 }
239
240 // The trie search breaks early when it hits allowed_mismatches, but
241 // allowed_mismatches might be different across calls to search(). If
242 // we break early and report a miss, the miss will be cached and
243 // returned in cases where there is a higher cap (and thus might
244 // actually be a hit). As such, we should only store a miss in the
245 // cache when the requested number of mismatches is equal to the
246 // maximum number of mismatches that was specified in the constructor.
247 //
248 // Of course, if the search failed because of ambiguity, then it would
249 // have failed even if we were searching with maximum mismatches;
250 // so we happily cache that.
251 if (allowed_mismatches == my_max_mm || missed.index == STATUS_AMBIGUOUS) {
252 state.cache[search_seq] = CacheEntry(missed.index, missed.mismatches);
253 }
254
255 state.index = missed.index;
256 state.mismatches = 0;
