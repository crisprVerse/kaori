23template<class Trie>
24void fill_library(
25 const std::vector<const char*>& options,
26 std::unordered_map<std::string, int>& exact,
27 Trie& trie,
28 bool reverse,
29 bool duplicates
30) {
31 size_t len = trie.get_length();
32
33 for (size_t i = 0; i < options.size(); ++i) {
34 auto ptr = options[i];
35
36 std::string current;
37 if (!reverse) {
38 current = std::string(ptr, ptr + len);
39 } else {
40 current.reserve(len);
41 for (size_t j = 0; j < len; ++j) {
42 current += reverse_complement<true, true>(ptr[len - j - 1]);
43 }
44 }
45
46 // Note that this must be called, even if the sequence is duplicated;
47 // otherwise the trie's internal counter will not be properly incremented.
48 bool has_iupac = trie.add(current.c_str(), duplicates);
49
50 if (!has_iupac) {
51 if (exact.find(current) != exact.end()) {
52 if (!duplicates) {
53 throw std::runtime_error("duplicate variable sequence '" + current + "'");
54 }
55 } else {
56 exact[current] = i;
57 }
58 }
59 }
60
61 trie.optimize();
62 return;
63}

65template<class Methods, class Cache, class Trie, class Result, class Mismatch>
66void matcher_in_the_rye(const std::string& x, const Cache& cache, const Trie& trie, Result& res, const Mismatch& mismatches, const Mismatch& max_mismatches) {
67 // Seeing if it's any of the caches; otherwise searching the trie.
68 auto cit = cache.find(x);
69 if (cit == cache.end()) {
70 auto lit = res.cache.find(x);
71 if (lit != res.cache.end()) {
72 Methods::update(res, lit->second, mismatches);
73
74 } else {
75 auto missed = trie.search(x.c_str(), mismatches);
76
77 // The trie search breaks early when it hits the mismatch cap,
78 // but the cap might be different across calls. If we break
79 // early and report a miss, the miss will be cached and
80 // returned in cases where there is a higher cap (and thus
81 // might actually be a hit). As such, we should only store a
82 // miss in the cache when the requested number of mismatches is
83 // equal to the maximum value specified in the constructor.
84 if (Methods::index(missed) >= 0 || mismatches == max_mismatches) {
85 res.cache[x] = missed;
86 }
87
88 // No need to pass the requested number of mismatches,
89 // as we explicitly searched for that in the trie.
90 Methods::update(res, missed);
91 }
92 } else {
93 Methods::update(res, cit->second, mismatches);
94 }
95 return;
96}
