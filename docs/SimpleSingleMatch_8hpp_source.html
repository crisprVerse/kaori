231 bool search_best(const char* read_seq, size_t read_length, State& state) const {

233 state.index = -1;
234 bool found = false;

237 auto update = [&](bool rev, int const_mismatches, const typename SimpleBarcodeSearch::State& x) -> void {
238 if (x.index < 0) {

240 }
241

243 if (total == best) {
244 if (state.index != x.index) { // ambiguous, setting back to a mismatch.
245 found = false;
246 state.index = -1;
247 }
248 } else if (total < best) {
