239 bool search_best(const char* read_seq, size_t read_length, State& state) const {
240 auto deets = constant.initialize(read_seq, read_length);
241 state.index = -1;
242 bool found = false;
243 int best = max_mm + 1;
244
245 auto update = [&](bool rev, int const_mismatches, const typename SimpleBarcodeSearch::State& x) -> void {
246 if (x.index < 0) {
247 return;
248 }
249
250 auto total = x.mismatches + const_mismatches;
251 if (total == best) {
252 if (state.index != x.index) { // ambiguous, setting back to a mismatch.
253 found = false;
254 state.index = -1;
255 }
256 } else if (total < best) {
257 found = true;
258 best = total;
259 // A further optimization at this point would be to narrow
260 // max_mm to the current 'best'. But this probably
261 // isn't worth it.

263 state.index = x.index;
264 state.mismatches = total;
265 state.variable_mismatches = x.mismatches;

268 }
