272 if (!is_barcode_index_ok(x.index)) {
273 return;
274 }
275
276 auto total = x.mismatches + const_mismatches;
277 if (total == best) {

279 found = false;

281 }
282 } else if (total < best) {
283 found = true;
284 best = total;

286 // max_mm to the current 'best'. But this probably
