260 bool search_best(const char* read_seq, size_t read_length, State& state) const {
261 auto deets = constant.initialize(read_seq, read_length);
262 state.index = -1;
263 bool found = false;
264 int best = max_mm + 1;
265
266 auto update = [&](bool rev, int const_mismatches, const typename SimpleBarcodeSearch::State& x) -> void {
267 if (x.index < 0) {
268 return;
269 }
270
271 auto total = x.mismatches + const_mismatches;
272 if (total == best) {
273 if (state.index != x.index) { // ambiguous, setting back to a mismatch.
274 found = false;
275 state.index = -1;
276 }
277 } else if (total < best) {
278 found = true;
279 best = total;
280 // A further optimization at this point would be to narrow
281 // max_mm to the current 'best'. But this probably
282 // isn't worth it.

284 state.index = x.index;
285 state.mismatches = total;
