215 bool process_best(State& state, const std::pair<const char*, const char*>& x) const {
216 auto deets = my_constant_matcher.initialize(x.first, x.second - x.first);
217 bool found = false;
218 int best_mismatches = my_max_mm + 1;
219 BarcodeIndex best_id = STATUS_UNMATCHED;
220
221 auto update = [&](std::pair<BarcodeIndex, int> match) -> void {
222 if (!is_barcode_index_ok(match.first)){
223 return;
224 }
225 if (match.second == best_mismatches) {
226 if (best_id != match.first) { // ambiguous.
227 found = false;
228 }
229 } else if (match.second < best_mismatches) {
230 // A further optimization at this point would be to narrow
231 // max_mm to the current 'best_mismatches'. But
232 // this probably isn't worth it.
233
234 found = true;
235 best_mismatches = match.second;
236 best_id = match.first;
237 }
238 };
239
240 while (!deets.finished) {
241 my_constant_matcher.next(deets);
242
243 if (my_forward && deets.forward_mismatches <= my_max_mm) {
244 update(forward_match(x.first, deets, state));
245 }
246
247 if (my_reverse && deets.reverse_mismatches <= my_max_mm) {
248 update(reverse_match(x.first, deets, state));
249 }
250 }
251
252 if (found) {
253 ++state.counts[best_id];
254 }
255 return found;
256 }
