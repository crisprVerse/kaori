127 void recursive_add(size_t i, int position, const char* barcode_seq, AddStatus& status) {
128 // Processing a stretch of non-ambiguous codes, where possible.
129 // This reduces the recursion depth among the (hopefully fewer) ambiguous codes.
130 while (1) {
131 auto shift = base_shift<true>(barcode_seq[i]);
132 if (shift == NON_STANDARD_BASE) {
133 break;
134 }
135
136 if ((++i) == length) {
137 end(shift, position, status);
138 return;
139 } else {
140 next(shift, position);
141 }
142 }
143
144 // Processing the ambiguous codes.
145 status.has_ambiguous = true;
146
147 auto process = [&](char base) -> void {
148 auto shift = base_shift(base);
149 if (i + 1 == length) {
150 end(shift, position, status);
151 } else {
152 auto curpos = position;
153 next(shift, curpos);
154 recursive_add(i + 1, curpos, barcode_seq, status);
155 }
156 };

158 switch(barcode_seq[i]) {
159 case 'R': case 'r':
160 process('A'); process('G'); break;
161 case 'Y': case 'y':
162 process('C'); process('T'); break;
163 case 'S': case 's':
164 process('C'); process('G'); break;
165 case 'W': case 'w':
166 process('A'); process('T'); break;
167 case 'K': case 'k':
168 process('G'); process('T'); break;
169 case 'M': case 'm':
170 process('A'); process('C'); break;
171 case 'B': case 'b':
172 process('C'); process('G'); process('T'); break;
173 case 'D': case 'd':
174 process('A'); process('G'); process('T'); break;
175 case 'H': case 'h':
176 process('A'); process('C'); process('T'); break;
177 case 'V': case 'v':
178 process('A'); process('C'); process('G'); break;
179 case 'N': case 'n':
180 process('A'); process('C'); process('G'); process('T'); break;
181 default:

275 // Optimization involves reorganizing the nodes so that the pointers are
276 // always increasing. This promotes memory locality of similar sequences
277 // in a depth-first search (which is what search() does anyway).
278 bool is_optimal(int node, size_t pos, int& maxed) const {
279 ++pos;
280 if (pos < length) {
281 for (int s = 0; s < NUM_BASES; ++s) {
282 auto v = pointers[node + s];
283 if (v < 0) {
284 continue;
285 }
286
287 if (v < maxed) {
288 return false;
289 }
290
291 maxed = v;
292 if (!is_optimal(v, pos, maxed)) {
293 return false;
294 }
295 }
296 }
297 return true;
298 }

300 void optimize(int node, size_t pos, std::vector<int>& trie) const {
301 auto it = pointers.begin() + node;
302 size_t new_node = trie.size();
303 trie.insert(trie.end(), it, it + NUM_BASES);
304
305 ++pos;
306 if (pos < length) {
307 for (int s = 0; s < NUM_BASES; ++s) {
308 auto& v = trie[new_node + s];
309 if (v < 0) {
310 continue;
311 }
312
313 auto original = v;
314 v = trie.size();
315 optimize(original, pos, trie);
316 }

362 std::pair<int, int> search(const char* seq, size_t pos, int node, int mismatches, int& max_mismatches) const {
363 int shift = base_shift<true>(seq[pos]);
364 int current = (shift >= 0 ? pointers[node + shift] : STATUS_MISSING);
365
366 // At the end: we prepare to return the actual values. We also refine
367 // the max number of mismatches so that we don't search for things with
368 // more mismatches than the best hit that was already encountered.
369 if (pos + 1 == length) {
370 if (current >= 0 || current == STATUS_AMBIGUOUS) {
371 max_mismatches = mismatches;
372 return std::make_pair(current, mismatches);
373 }
374
375 int alt = STATUS_MISSING;
376 ++mismatches;
377 if (mismatches <= max_mismatches) {
378 bool found = false;
379 for (int s = 0; s < NUM_BASES; ++s) {
380 if (shift == s) {
381 continue;
382 }
383
384 int candidate = pointers[node + s];
385 if (candidate >= 0) {
386 if (found) {
387 if (candidate != alt) { // protect against multiple occurrences of IUPAC code-containing barcodes.
388 if (duplicates == DuplicateAction::FIRST) {
389 if (candidate < alt) {
390 alt = candidate;
391 }
392 } else if (duplicates == DuplicateAction::LAST) {
393 if (candidate > alt) {
394 alt = candidate;
395 }
396 } else {
397 alt = STATUS_AMBIGUOUS; // ambiguous so we quit early.
398 break;
399 }
400 }
401 } else {
402 alt = candidate;
403 max_mismatches = mismatches;
404 found = true;
405 }
406
407 } else if (candidate == STATUS_AMBIGUOUS) {
408 // If an ambiguity is present on a base at the last position,
409 // and we're accepting a mismatch on the last position, then
410 // we already have at least two known barcodes that match the
411 // input sequence. The behavior of the other bases is irrelevant;
412 // even if they refer to non-ambiguous barcodes, that just adds to the
413 // set of 2+ barcodes that the input sequence already matches.
414 // So, we have no choice but to fail the match due to ambiguity.
415 alt = STATUS_AMBIGUOUS;
416 max_mismatches = mismatches;
417 break;
418 }
419 }
420 }
421
422 return std::make_pair(alt, mismatches);
423
424 } else {
425 ++pos;
426
427 std::pair<int, int> best(STATUS_MISSING, max_mismatches + 1);
428 if (current >= 0) {
429 best = search(seq, pos, current, mismatches, max_mismatches);
430 }
431
432 ++mismatches;
433 if (mismatches <= max_mismatches) {
434 for (int s = 0; s < NUM_BASES; ++s) {
435 if (shift == s) {
436 continue;
437 }
438
439 int alt = pointers[node + s];
440 if (alt < 0) {
441 continue;
442 }
443
444 auto chosen = search(seq, pos, alt, mismatches, max_mismatches);
445 if (chosen.first >= 0) {
446 if (chosen.second < best.second) {
447 best = chosen;
448 } else if (chosen.second == best.second) {
449 if (chosen.first != best.first) { // protect against multiple occurrences of IUPAC code-containing barcodes.
450 if (duplicates == DuplicateAction::FIRST) {
451 if (chosen.first < best.first) {
452 best.first = chosen.first;
453 }
454 } else if (duplicates == DuplicateAction::LAST) {
455 if (chosen.first > best.first) {
456 best.first = chosen.first;
457 }
458 } else {
459 best.first = STATUS_AMBIGUOUS;
460 }
461 }
462 }
463
464 } else if (chosen.first == STATUS_AMBIGUOUS) {
465 if (chosen.second < best.second) {
466 best = chosen;
467 } else if (chosen.second == best.second) {
468 // Ambiguity is infectious, as described above.
469 best.first = STATUS_AMBIGUOUS;
470 }
471 }
472 }
473 }
474
475 return best;
