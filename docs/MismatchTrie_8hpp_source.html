52int trie_base_shift() {
53 if constexpr(base_ == 'A') {
54 return 0;
55 } else if constexpr(base_ == 'C') {
56 return 1;
57 } else if constexpr(base_ == 'G') {
58 return 2;
59 } else { // i.e., base_ == 'T'
60 return 3;
61 }

133 void recursive_add(SeqLength i, BarcodeIndex node, const char* barcode_seq, TrieAddStatus& status) {
134 // Processing a stretch of non-ambiguous codes, where possible.
135 // This reduces the recursion depth among the (hopefully fewer) ambiguous codes.
136 while (1) {
137 switch (barcode_seq[i]) {
138 case 'A': case 'a':
139 node += trie_base_shift<'A'>(); break;
140 case 'C': case 'c':
141 node += trie_base_shift<'C'>(); break;
142 case 'G': case 'g':
143 node += trie_base_shift<'G'>(); break;
144 case 'T': case 't':
145 node += trie_base_shift<'T'>(); break;
146 default:
147 goto ambiguous;
148 }
149 if ((++i) == my_length) {
150 end(node, status);
151 return;
152 } else {
153 node = next(node);
154 }
155 }
156
157ambiguous:
158 // Processing the ambiguous codes.
159 status.has_ambiguous = true;
160
161 auto processA = [&]() -> void { process_ambiguous<'A'>(i, node, barcode_seq, status); };
162 auto processC = [&]() -> void { process_ambiguous<'C'>(i, node, barcode_seq, status); };
163 auto processG = [&]() -> void { process_ambiguous<'G'>(i, node, barcode_seq, status); };
164 auto processT = [&]() -> void { process_ambiguous<'T'>(i, node, barcode_seq, status); };
165
166 switch(barcode_seq[i]) {
167 case 'R': case 'r':
168 processA(); processG(); break;
169 case 'Y': case 'y':
170 processC(); processT(); break;
171 case 'S': case 's':
172 processC(); processG(); break;
173 case 'W': case 'w':
174 processA(); processT(); break;
175 case 'K': case 'k':
176 processG(); processT(); break;
177 case 'M': case 'm':
178 processA(); processC(); break;
179 case 'B': case 'b':
180 processC(); processG(); processT(); break;
181 case 'D': case 'd':
182 processA(); processG(); processT(); break;
183 case 'H': case 'h':
184 processA(); processC(); processT(); break;
185 case 'V': case 'v':
186 processA(); processC(); processG(); break;
187 case 'N': case 'n':
188 processA(); processC(); processG(); processT(); break;
189 default:
190 throw std::runtime_error("unknown base '" + std::string(1, barcode_seq[i]) + "' detected when constructing the trie");
191 }
192 }

217 void replace_best_with_chosen(SearchResult_& best, BarcodeIndex& best_index, int best_score, const SearchResult_& chosen, BarcodeIndex chosen_index, int chosen_score) const {
218 if (is_barcode_index_ok(chosen_index)) {
219 if (chosen_score < best_score) {
220 best = chosen;
221 } else if (chosen_score == best_score) {
222 if (chosen_index != best_index) { // protect against multiple occurrences of IUPAC code-containing barcodes.
223 if (my_duplicates == DuplicateAction::FIRST) {
224 if (chosen_index < best_index) {
225 best_index = chosen_index;
226 }
227 } else if (my_duplicates == DuplicateAction::LAST) {
228 if (chosen_index > best_index) {
229 best_index = chosen_index;
230 }
231 } else {
232 best_index = STATUS_AMBIGUOUS;
233 }
234 }
235 }
236
237 } else if (chosen_index == STATUS_AMBIGUOUS) {
238 if (chosen_score < best_score) {
239 best = chosen;
240 } else if (chosen_score == best_score) {
241 // Ambiguity is infectious. Each ambiguous status indicates that there
242 // are already 2+ barcodes on this score, so it doesn't matter how
243 // many other unambiguous barcodes are here; we're already ambiguous.

249 // To be called in the last step of the recursive search.
250 void scan_final_position_with_mismatch(BarcodeIndex node, int refshift, BarcodeIndex& current_index, int current_mismatches, int& mismatch_cap) const {
251 bool found = false;
252 for (int s = 0; s < NUM_BASES; ++s) {
253 if (s == refshift) {
254 continue;
255 }
256
257 auto candidate = my_pointers[node + s];
258 if (is_barcode_index_ok(candidate)) {
259 if (found) {
260 if (candidate != current_index) { // protect against multiple occurrences of IUPAC-containg barcodes.
261 if (my_duplicates == DuplicateAction::FIRST) {
262 if (current_index > candidate) {
263 current_index = candidate;
264 }
265 } else if (my_duplicates == DuplicateAction::LAST) {
266 if (current_index < candidate) {
267 current_index = candidate;
268 }
269 } else {
270 current_index = STATUS_AMBIGUOUS; // ambiguous, so we quit early.
271 break;
272 }
273 }
274 } else {
275 current_index = candidate;
276 mismatch_cap = current_mismatches;
277 found = true;
278 }
279
280 } else if (candidate == STATUS_AMBIGUOUS) {
281 // If an ambiguity is present on a base at the last position,
282 // and we're accepting a mismatch on the last position, then
283 // we already have at least two known barcodes that match the
284 // input sequence. The behavior of the other bases is irrelevant;
285 // even if they refer to non-ambiguous barcodes, that just adds to the
286 // set of 2+ barcodes that the input sequence already matches.
287 // So, we have no choice but to fail the match due to ambiguity.
288 current_index = STATUS_AMBIGUOUS;
289 mismatch_cap = current_mismatches;
290 break;
291 }
292 }

307 // Optimization involves reorganizing the nodes so that the pointers are
308 // always increasing. This promotes memory locality of similar sequences
309 // in a depth-first search (which is what search() does anyway).
310 bool is_optimal(SeqLength i, BarcodeIndex node, BarcodeIndex& maxed) const {
311 ++i;
312 if (i < my_length) {
313 for (int s = 0; s < NUM_BASES; ++s) {
314 auto v = my_pointers[node + s];
315 if (!is_barcode_index_ok(v)) {
316 continue;
317 }
318
319 if (v < maxed) {
320 return false;
321 }
322
323 maxed = v;
324 if (!is_optimal(i, v, maxed)) {
325 return false;
326 }
327 }
328 }
329 return true;
330 }

334 BarcodeIndex new_node = trie.size();
335 trie.insert(trie.end(), it, it + NUM_BASES);
336
337 ++i;

339 for (int s = 0; s < NUM_BASES; ++s) {
340 auto& v = trie[new_node + s];
341 if (!is_barcode_index_ok(v)) {
342 continue;
343 }

345 auto original = v;

353inline std::pair<BarcodeIndex, int> trie_next_base(char base, BarcodeIndex node, const std::vector<BarcodeIndex>& pointers) {
354 BarcodeIndex current;
355 int shift;
356 switch (base) {
357 case 'A': case 'a':
358 shift = trie_base_shift<'A'>(); current = pointers[node + shift]; break;
359 case 'C': case 'c':
360 shift = trie_base_shift<'C'>(); current = pointers[node + shift]; break;
361 case 'G': case 'g':
362 shift = trie_base_shift<'G'>(); current = pointers[node + shift]; break;
363 case 'T': case 't':
364 shift = trie_base_shift<'T'>(); current = pointers[node + shift]; break;
365 default:
366 shift = -1; current = STATUS_UNMATCHED; break;
367 }
368 return std::make_pair(current, shift);

473 Result search(const char* seq, SeqLength i, BarcodeIndex node, int mismatches, int& max_mismatches) const {
474 const auto& pointers = my_core.pointers();
475 auto next = trie_next_base(seq[i], node, pointers);
476 auto current = next.first;
477 auto shift = next.second;
478
479 // At the end: we prepare to return the actual values. We also refine
480 // the max number of mismatches so that we don't search for things with
481 // more mismatches than the best hit that was already encountered.
482 SeqLength length = my_core.length();
483 ++i;
484
485 if (i == length) {
486 if (is_barcode_index_ok(current) || current == STATUS_AMBIGUOUS) {
487 max_mismatches = mismatches; // this assignment should always decrease max_mismatches, otherwise the search would have terminated earlier.
488 return Result(current, mismatches);
489 }
490
491 BarcodeIndex alt = STATUS_UNMATCHED;
492 ++mismatches;
493 if (mismatches <= max_mismatches) {
494 my_core.scan_final_position_with_mismatch(node, shift, alt, mismatches, max_mismatches);
495 }
496
497 return Result(alt, mismatches);
498
499 } else {
500 Result best(STATUS_UNMATCHED, max_mismatches + 1);
501 if (is_barcode_index_ok(current)) {
502 best = search(seq, i, current, mismatches, max_mismatches);
503 }
504
505 ++mismatches;
506 if (mismatches <= max_mismatches) {
507 for (int s = 0; s < NUM_BASES; ++s) {
508 if (shift == s) {
509 continue;
510 }
511
512 auto alt = pointers[node + s];
513 if (!is_barcode_index_ok(alt)) {
514 continue;
515 }
516
517 if (mismatches <= max_mismatches) { // check again, just in case max_mismatches changed.
518 auto chosen = search(seq, i, alt, mismatches, max_mismatches);
519 my_core.replace_best_with_chosen(best, best.index, best.mismatches, chosen, chosen.index, chosen.mismatches);
520 }
521 }

654 // as the index of the node on the trie.
655 auto node = state.index;
656
657 const auto& pointers = my_core.pointers();

663 // the max number of mismatches so that we don't search for things with
