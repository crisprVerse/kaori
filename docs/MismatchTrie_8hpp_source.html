88 void end(int shift, int position, DuplicateAction duplicates, AddStatus& status) {

115 void recursive_add(size_t i, int position, const char* barcode_seq, DuplicateAction duplicates, AddStatus& status) {
116 // Processing a stretch of non-ambiguous codes, where possible.
117 // This reduces the recursion depth among the (hopefully fewer) ambiguous codes.
118 while (1) {
119 auto shift = base_shift<true>(barcode_seq[i]);
120 if (shift == -1) {
121 break;
122 }
123
124 if ((++i) == length) {
125 end(shift, position, duplicates, status);
126 return;
127 } else {
128 next(shift, position);
129 }
130 }
131
132 // Processing the ambiguous codes.
133 status.has_ambiguous = true;
134
135 auto process = [&](char base) -> void {
136 auto shift = base_shift(base);
137 if (i + 1 == length) {
138 end(shift, position, duplicates, status);
139 } else {
140 auto curpos = position;
141 next(shift, curpos);
142 recursive_add(i + 1, curpos, barcode_seq, duplicates, status);
143 }
144 };
145
146 switch(barcode_seq[i]) {
147 case 'R': case 'r':
148 process('A'); process('G'); break;
149 case 'Y': case 'y':
150 process('C'); process('T'); break;
151 case 'S': case 's':
152 process('C'); process('G'); break;
153 case 'W': case 'w':
154 process('A'); process('T'); break;
155 case 'K': case 'k':
156 process('G'); process('T'); break;
157 case 'M': case 'm':
158 process('A'); process('C'); break;
159 case 'B': case 'b':
160 process('C'); process('G'); process('T'); break;
161 case 'D': case 'd':
162 process('A'); process('G'); process('T'); break;
163 case 'H': case 'h':
164 process('A'); process('C'); process('T'); break;
165 case 'V': case 'v':
166 process('A'); process('C'); process('G'); break;
167 case 'N': case 'n':
168 process('A'); process('C'); process('G'); process('T'); break;
169 default:

171 }

260 // Optimization involves reorganizing the nodes so that the pointers are
261 // always increasing. This promotes memory locality of similar sequences
262 // in a depth-first search (which is what search() does anyway).
263 bool is_optimal(int node, size_t pos, int& maxed) const {
264 ++pos;
265 if (pos < length) {
266 for (int s = 0; s < 4; ++s) {
267 auto v = pointers[node + s];
268 if (v < 0) {
269 continue;
270 }
271
272 if (v < maxed) {
273 return false;
274 }
275
276 maxed = v;
277 if (!is_optimal(v, pos, maxed)) {
278 return false;
279 }
280 }
281 }
282 return true;
283 }

285 void optimize(int node, size_t pos, std::vector<int>& trie) const {

287 size_t new_node = trie.size();
288 trie.insert(trie.end(), it, it + 4);
289
290 ++pos;
291 if (pos < length) {
292 for (int s = 0; s < 4; ++s) {
293 auto& v = trie[new_node + s];
294 if (v < 0) {
295 continue;
296 }

298 auto original = v;
299 v = trie.size();

345 int current = (shift >= 0 ? pointers[node + shift] : -1);
346
347 // At the end: we prepare to return the actual values. We also refine
348 // the max number of mismatches so that we don't search for things with
349 // more mismatches than the best hit that was already encountered.
350 if (pos + 1 == length) {
351 if (current >= 0) {
352 max_mismatches = mismatches;
353 return std::make_pair(current, mismatches);
354 }
355
356 int alt = -1;
357 ++mismatches;
358 if (mismatches <= max_mismatches) {
359 bool found = false;
360 for (int s = 0; s < 4; ++s) {
361 if (shift == s) {
362 continue;
363 }
364

366 if (candidate >= 0) {
367 if (found) { // ambiguous, so we quit early.
368 alt = -1;
369 break;
370 }
371 alt = candidate;
372 max_mismatches = mismatches;
373 found = true;
374 }

378
379 } else {
380 ++pos;
381

383 if (current >= 0) {
384 best = search(seq, pos, current, mismatches, max_mismatches);
