151 // Optimization involves reorganizing the nodes so that the pointers are
152 // always increasing. This promotes memory locality of similar sequences
153 // in a depth-first search (which is what search() does anyway).
154 bool is_optimal(int node, size_t pos, int& maxed) const {
155 ++pos;
156 if (pos < length) {
157 for (int s = 0; s < 4; ++s) {
158 auto v = pointers[node + s];
159 if (v < 0) {
160 continue;
161 }
162
163 if (v < maxed) {
164 return false;
165 }
166
167 maxed = v;
168 if (!is_optimal(v, pos, maxed)) {
169 return false;
170 }
171 }
172 }
173 return true;
174 }
